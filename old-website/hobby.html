SELECT
         -- ATTRIBUTES
         reporting_starts AS logdate,
         CASE
                  WHEN impression_device LIKE "%ipad%"
                  OR       impression_device LIKE "%iphone%"
                  OR       impression_device LIKE "%ipod%"
                  OR       impression_device LIKE "%smartphone%"
                  OR       impression_device LIKE "%tablet%"
                  OR       impression_device LIKE "%blackberry%"
                  OR       impression_device LIKE "%android%"
                  OR       impression_device LIKE "%mobile%" THEN "Mobile"
                  WHEN impression_device LIKE "%computer%"
                  OR       impression_device LIKE "%desktop%"
                  OR       impression_device LIKE "%console%"
                  OR       impression_device LIKE "%linux%"
                  OR       impression_device LIKE "%vista%"
                  OR       impression_device LIKE "%xp%"
                  OR       impression_device LIKE "%windows%"
                  OR       impression_device LIKE "%APPLE_OS_X%" THEN "Desktop"
                  ELSE "Unknown"
         END           AS device,
         'Facebook'    AS publisher,
         'High Volume' AS partners_by_volume,
         CASE
                  WHEN ad_set_name LIKE "%_EN%" THEN "EN"
                  WHEN ad_set_name LIKE "%_FR%" THEN "FR"
                  WHEN ad_set_name LIKE "%_EFL%" THEN "All Languages"
                  ELSE "NA"
         END      AS language,
         'Social' AS channel,
         CASE
                  WHEN ad_name LIKE "%BTS%" THEN "BTS Event"
                  WHEN ad_name LIKE "%school%" THEN "BTS Seasonal"
                  WHEN ad_name LIKE "%Kitchen%" THEN "Kitchen Event"
                  WHEN ad_name LIKE "%Bed%" THEN "Bedroom Event"
                  WHEN ad_name LIKE "%Wardrobe%" THEN "Wardrobe Event"
                  WHEN ad_name LIKE "%Dining%" THEN "Dining Event"
                  WHEN ad_name LIKE "%Bathroom%" THEN "Bathroom Event"
                  WHEN ad_name LIKE "%AlwaysOnKtch%" THEN "Kitchen Always On"
                  WHEN ad_name LIKE "%AlwaysOnBdrm%" THEN "Bedroom Always On"
                  WHEN ad_name LIKE "%AlwaysOnWrdr%" THEN "Wardrobe Always On"
                  WHEN ad_name LIKE "%AlwaysOnDnng%" THEN "Dining Always On"
                  WHEN ad_name LIKE "%AlwaysOnBthr%" THEN "Bathroom Always On"
                  ELSE "NA_facebook"
         END AS campaign,
         CASE
                  WHEN campaign_name LIKE "%prospect%" THEN "Prospecting"
                  WHEN campaign_name LIKE "%remarketing%" THEN "Remarketing"
                  WHEN campaign_name LIKE "%similar%" THEN "Prospecting"
                  WHEN campaign_name LIKE "%campaign_name%" THEN "Remarketing" -- naming
                           convention was NOT followed
                  WHEN campaign_name LIKE "%ikea_roi_soc_wk24bathroomevent%" THEN "Remarketing" -- naming convention was not followed
                  ELSE "NA_facebook"
         END AS tactic,
         --METRICS
         sum(amount_spent_cad)                                                AS spend,
         sum(impressions)                                                     AS impressions,
         sum(link_clicks_28_days_after_clicking)                              AS clicks,
         sum(purchase_facebook_pixel_28_days_after_clicking)                  AS orders_ct,
         sum(purchase_facebook_pixel_7_days_after_viewing)                    AS orders_vt,
         sum(purchase_conversion_value_facebook_pixel_28_days_after_clicking) AS revenue_ct,
         sum(purchase_conversion_value_facebook_pixel_7_days_after_viewing)   AS revenue_vt,
         0                                                                    AS store_visits
FROM     [TestProject:TestDataSet.TestTable]
GROUP BY logdate,
         device,
         publisher,
         partners_by_volume,
         language,
         channel,
         campaign,
         tactic ignore
         CASE
